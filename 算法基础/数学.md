# [[信息与未来 2018] 最大公约数](https://www.luogu.com.cn/problem/B3736)
```
#include <bits/stdc++.h>

using namespace std;

int x, y, z;

int gcd(int a, int b) {
	return b == 0 ? a : gcd(b, a % b);
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0), cout.tie(0);
	
	cin >> x >> y >> z;
	
	cout << gcd(x, gcd(y, z)) << endl;
	
	return 0;
} 
```
