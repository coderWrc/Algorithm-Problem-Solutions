```
#include <iostream>

using namespace std;

const int N = 25;

int n, k;
int a[N];
int sum;
int ans;

bool is_prime(int x)
{
	if (x == 1) return false;

	for (int i = 2; i <= x / i; i++)
	{
		if (x % i == 0) return false;
	}
	return true;
}

void dfs(int pos, int begin)
{
	if (pos > k)
	{
		if (is_prime(sum)) ans++;
		return;
	}

	for (int i = begin; i <= n; i++)
	{
		sum += a[i];
		dfs(pos + 1, i + 1);
		sum -= a[i];
	}
}

int main()
{
	cin >> n >> k;
	for (int i = 1; i <= n; i++) cin >> a[i];
	dfs(1, 1);
	cout << ans << endl;

	return 0;
}
```

```
#include <iostream>
#include <cstring>

using namespace std;

const int N = 15;

int n;
int t[N], d[N], l[N]; // 起始时间 盘旋时间 降落时间
bool st[N];

bool dfs(int pos, int end)
{
	// 如果前 n 个都可以
	if (pos > n) return true;

	for (int i = 1; i <= n; i++)
	{
		if (st[i] || end > t[i] + d[i]) continue;
		st[i] = true;
		int nextend = max(end, t[i]) + l[i];
		if (dfs(pos + 1, nextend)) return true;
		st[i] = false;
	}
	return false;
}

int main()
{
	int T; cin >> T;
	while (T--)
	{
		memset(st, 0, sizeof st);
		cin >> n;
		for (int i = 1; i <= n; i++) cin >> t[i] >> d[i] >> l[i];

		if (dfs(1, 0)) cout << "YES" << endl;
		else cout << "NO" << endl;
	}


	return 0;
}
```

```
#include <iostream>
#include <vector>

using namespace std;

const int N = 15;

int n;
bool col[N], st1[N << 1], st2[N << 1];

vector<int> path;
int ans;

void dfs(int x)
{
	if (x > n)
	{
		ans++;
		if (ans <= 3)
		{
			for (int x : path) cout << x << ' ';
			cout << endl;
		}
	}

	for (int y = 1; y <= n; y++)
	{
		// 判断能不能摆在这一列
		if (col[y] || st1[x + y] || st2[y - x + n]) continue; // 剪枝
		col[y] = st1[x + y] = st2[y - x + n] = true;
		path.push_back(y);
		dfs(x + 1);
		col[y] = st1[x + y] = st2[y - x + n] = false;
		path.pop_back();
	}
}

int main()
{
	cin >> n;
	dfs(1);
	cout << ans << endl;

	return 0;
```

# [数独](https://www.luogu.com.cn/problem/P1784)
[top](#1-深度优先搜索---dfs)
```
#include <iostream>

using namespace std;

const int N = 10;

int n = 9;
int a[N][N];
bool row[N][N], col[N][N], st[N][N][N];

bool dfs(int i, int j)
{
	// 如果这一行填满了
	if (j == n)
	{
		i++;
		j = 0;
	}

	// 找到一种合法的情况，就停止递归
	if (i == n) return true;
	// 如果一开始这里就有数，就看下一个格子
	if (a[i][j]) return dfs(i, j + 1);

	for (int x = 1; x <= 9; x++)
	{
		if (row[i][x] || col[j][x] || st[i / 3][j / 3][x]) continue; // 剪枝

		row[i][x] = col[j][x] = st[i / 3][j / 3][x] = true;
		a[i][j] = x;

		if (dfs(i, j + 1)) return true;

		// 恢复现场
		row[i][x] = col[j][x] = st[i / 3][j / 3][x] = false;
		a[i][j] = 0;
	}
	return false;
}

int main()
{
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			cin >> a[i][j];
			int x = a[i][j];
			if (x) row[i][x] = col[j][x] = st[i / 3][j / 3][x] = true;
		}
	}

	dfs(0, 0);

	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			cout << a[i][j] << ' ';
		}
		cout << endl;
	}

	return 0;
}
```

# [[NOIP 2001 提高组] 数的划分](https://www.luogu.com.cn/problem/P1025)
[top](#1-深度优先搜索---dfs)
```
#include <iostream>

using namespace std;

int n, k;
int path, ans;

void dfs(int pos, int begin)
{
	if (pos == k)
	{
		if (path == n) ans++;
		return;
	}

	for (int i = begin; i <= n; i++)
	{
		if (path + i * (k - pos) > n) return;

		path += i;
		dfs(pos + 1, i);
		path -= i;
	}
}

int main()
{
	cin >> n >> k;
	dfs(0, 1);
	cout << ans << endl;

	return 0;
}
```

# [小猫爬山](https://www.luogu.com.cn/problem/P10483)
[top](#1-深度优先搜索---dfs)
```
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 20;

int n, w;
int c[N];

int cnt; // 当前用了多少辆车
int s[N]; // 每辆车目前的总重
int ans = N; // 最优解

bool cmp(int a, int b)
{
	return a > b;
}

void dfs(int pos)
{
	if (cnt >= ans) return;

	if (pos > n)
	{
		ans = cnt;
		return;
	}

	// 先分配质量大的猫

	// 看能不能不新开车
	for (int i = 1; i <= cnt; i++)
	{
		if (c[pos] + s[i] > w) continue;

		s[i] += c[pos];
		dfs(pos + 1);
		s[i] -= c[pos];
	}

	// 新开一辆车
	s[++cnt] = c[pos];
	dfs(pos + 1);
	s[cnt--] = 0;
}

int main()
{
	cin >> n >> w;
	for (int i = 1; i <= n; i++) cin >> c[i];
	sort(c + 1, c + 1 + n, cmp);
	dfs(1);
	cout << ans << endl;

	return 0;
}
```

[top](#1-深度优先搜索---dfs)

```
class Solution 
{
    int f[31];
public:
    int fib(int n) 
    {
        memset(f, -1, sizeof f);

        return dfs(n);
    }

    int dfs(int n)
    {
        if (f[n] != -1) return f[n];

        if (n < 2) return n;

        f[n] = dfs(n - 1) + dfs(n - 2);
        return f[n];
    }
};
```

```
#include <iostream>

using namespace std;

const int N = 110;

int r, c;
int a[N][N];
int f[N][N];

int dx[4] = { 0, 0, 1, -1 };
int dy[4] = { 1, -1, 0, 0 };

int dfs(int i, int j)
{
	if (f[i][j]) return f[i][j];

	int len = 1;

	for (int k = 0; k < 4; k++)
	{
		int x = i + dx[k], y = j + dy[k];
		if (x < 1 || x > r || y < 1 || y > c || a[i][j] <= a[x][y]) continue;
		len = max(len, dfs(x, y) + 1);
	}
	return f[i][j] = len;
}

int main()
{
	cin >> r >> c;
	for (int i = 1; i <= r; i++)
		for (int j = 1; j <= c; j++)
			cin >> a[i][j];

	int ans = 1;
	for (int i = 1; i <= r; i++)
		for (int j = 1; j <= c; j++)
			ans = max(ans, dfs(i, j));

	cout << ans << endl;

	return 0;
}
```

# [马的遍历](https://www.luogu.com.cn/problem/P1443)

```
#include <iostream>
#include <cstring>
#include <queue>

using namespace std;

typedef pair<int, int> PII;

const int N = 410;

int n, m, x, y;

int dist[N][N];

int dx[8] = {1, 2, 2, 1, -1, -2, -2, -1};
int dy[8] = {2, 1, -1, -2, -2, -1, 1, 2};

void bfs()
{
	memset(dist, -1, sizeof dist);

	queue<PII> q;
	q.push({ x, y });
	dist[x][y] = 0;

	while (q.size())
	{
		auto t = q.front(); q.pop();
		int i = t.first, j = t.second;
		for (int k = 0; k < 8; k++)
		{
			int x = i + dx[k], y = j + dy[k];
			if (x < 1 || x > n || y < 1 || y > m || dist[x][y] != -1) continue;
			dist[x][y] = dist[i][j] + 1;
			q.push({ x, y });
		}
	}
}

int main()
{
	cin >> n >> m >> x >> y;

	bfs();

	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			cout << dist[i][j] << ' ';
		}
		cout << endl;
	}

	return 0;
}
```

```
#include <cstring>
#include <queue>

using namespace std;

typedef pair<int, int> PII;

const int N = 35;

int n, m, x, y;
char a[N][N];
int dist[N][N];

int dx[4] = {0, 0, 1, -1};
int dy[4] = {1, -1, 0, 0};

void bfs()
{
    memset(dist, -1, sizeof dist);
    
    queue<PII> q;
    q.push({x, y});
    dist[x][y] = 0;
    while (q.size())
    {
        auto t = q.front(); q.pop();
        int i = t.first, j = t.second;
        for (int k = 0; k < 4; k++)
        {
            int x = i + dx[k], y = j + dy[k];
            if (a[x][y] != '.' && a[x][y] != 'e' || dist[x][y] != -1) continue;
            {
                dist[x][y] = dist[i][j] + 1;

        }
    }
}

    {
        for (int j = 1; j <= m; j++)
        {

            }
        }

    {
        for (int j = 1; j <= m; j++)
        {

            {
                cnt++;
                ans = min(ans, dist[i][j]);
            }
        }
    }
```

```
	queue<int> q;

		int a = t + 1, b = t - 1, c = t * 2;
```

# [八数码难题](https://www.luogu.com.cn/problem/P1379)
[top](#2-宽度优先搜索---bfs)
```
#include <iostream>
#include <unordered_map>
#include <queue>

using namespace std;

string s;
string aim = "123804765";

unordered_map<string, int> dist;

int dx[4] = { 0, 0, 1, -1 };
int dy[4] = { 1, -1, 0, 0 };

void bfs()
{
	queue<string> q;
	q.push(s);
	dist[s] = 0;

	while (q.size())
	{
		string t = q.front(); q.pop();

		// 找到空格 0 的位置
		int pos = 0;
		while (t[pos] != '0') pos++; // 注意这里不能是数字0
		int i = pos / 3, j = pos % 3;

		for (int k = 0; k < 4; k++)
		{
			int x = i + dx[k], y = j + dy[k];

			if (x < 0 || x > 2 || y < 0 || y > 2) continue;

			string next = t;
			int p = x * 3 + y;
			swap(next[pos], next[p]);
			if (dist.count(next)) continue;

			dist[next] = dist[t] + 1;
			if (next == aim) return; // 剪枝
```
