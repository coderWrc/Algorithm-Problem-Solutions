# [枚举子集（递归实现指数型枚举）](https://www.luogu.com.cn/problem/B3622)  
[top](#1-深度优先搜索---dfs)
```
#include <iostream>

using namespace std;

int n;
string path;

void dfs(int pos)
{
	if (pos > n)
	{
		cout << path << endl;
		return;
	}
	path += 'N';
	dfs(pos + 1);
	path.pop_back();
	path += 'Y';
	dfs(pos + 1);
	path.pop_back();
}

int main()
{
	cin >> n;
	dfs(1);

	return 0;
}
```

# [组合型枚举](https://www.luogu.com.cn/problem/P10448)

```
#include <iostream>
#include <vector>

using namespace std;

int n, m;
vector<int> path;

void dfs(int begin)
{
	if (path.size() == m)
	{
		for (auto x : path)
		{
			cout << x << ' ';
		}
		cout << endl;
		return;
	}
	for (int i = begin; i <= n; i++)
	{
		path.push_back(i);
		dfs(i + 1);
		path.pop_back();
	}
}

int main()
{
	cin >> n >> m;
	dfs(1);

	return 0;
}
```

# [枚举排列（递归实现排列型枚举）](https://www.luogu.com.cn/problem/B3623)
[top](#1-深度优先搜索---dfs)
```
#include <iostream>
#include <vector>

using namespace std;

int n, k;
vector<int> path;
bool st[11]; // 记录哪些学生已经被选过了

void dfs()
{
	if (path.size() == k)
	{
		for (int x : path) cout << x << ' ';
		cout << endl;
		return;
	}
	for (int i = 1; i <= n; i++)
	{
		if (st[i]) continue; // 如果已经选过了，就跳过
		st[i] = true; // 选他
		path.push_back(i);
		dfs();
		st[i] = false;
		path.pop_back();
	}
}

int main()
{
	cin >> n >> k;
	dfs();

	return 0;
}
```

# [[NOIP 2002 普及组] 选数](https://www.luogu.com.cn/problem/P1036)
[top](#1-深度优先搜索---dfs)
```
#include <iostream>

using namespace std;

const int N = 25;

int n, k;
int a[N];
int sum;
int ans;

bool is_prime(int x)
{
	if (x == 1) return false;

	for (int i = 2; i <= x / i; i++)
	{
		if (x % i == 0) return false;
	}
	return true;
}

void dfs(int pos, int begin)
{
	if (pos > k)
	{
		if (is_prime(sum)) ans++;
		return;
	}

	for (int i = begin; i <= n; i++)
	{
		sum += a[i];
		dfs(pos + 1, i + 1);
		sum -= a[i];
	}
}

int main()
{
	cin >> n >> k;
	for (int i = 1; i <= n; i++) cin >> a[i];
	dfs(1, 1);
	cout << ans << endl;

	return 0;
}
```
# [[蓝桥杯 2023 省 B] 飞机降落](https://www.luogu.com.cn/problem/P9241)
[top](#1-深度优先搜索---dfs)
```
#include <iostream>
#include <cstring>

using namespace std;

const int N = 15;

int n;
int t[N], d[N], l[N]; // 起始时间 盘旋时间 降落时间
bool st[N];

bool dfs(int pos, int end)
{
	// 如果前 n 个都可以
	if (pos > n) return true;

	for (int i = 1; i <= n; i++)
	{
		if (st[i] || end > t[i] + d[i]) continue;
		st[i] = true;
		int nextend = max(end, t[i]) + l[i];
		if (dfs(pos + 1, nextend)) return true;
		st[i] = false;
	}
	return false;
}

int main()
{
	int T; cin >> T;
	while (T--)
	{
		memset(st, 0, sizeof st);
		cin >> n;
		for (int i = 1; i <= n; i++) cin >> t[i] >> d[i] >> l[i];

		if (dfs(1, 0)) cout << "YES" << endl;
		else cout << "NO" << endl;
	}


	return 0;
}
```
# [[USACO1.5] 八皇后 Checker Challenge](https://www.luogu.com.cn/problem/P1219)
[top](#1-深度优先搜索---dfs)
```
#include <iostream>
#include <vector>

using namespace std;

const int N = 15;

int n;
bool col[N], st1[N << 1], st2[N << 1];

vector<int> path;
int ans;

void dfs(int x)
{
	if (x > n)
	{
		ans++;
		if (ans <= 3)
		{
			for (int x : path) cout << x << ' ';
			cout << endl;
		}
	}

	for (int y = 1; y <= n; y++)
	{
		// 判断能不能摆在这一列
		if (col[y] || st1[x + y] || st2[y - x + n]) continue; // 剪枝
		col[y] = st1[x + y] = st2[y - x + n] = true;
		path.push_back(y);
		dfs(x + 1);
		col[y] = st1[x + y] = st2[y - x + n] = false;
		path.pop_back();
	}
}

int main()
{
	cin >> n;
	dfs(1);
	cout << ans << endl;

	return 0;
}
```
# [数独](https://www.luogu.com.cn/problem/P1784)
[top](#1-深度优先搜索---dfs)
```
#include <iostream>

using namespace std;

const int N = 10;

int n = 9;
int a[N][N];
bool row[N][N], col[N][N], st[N][N][N];

bool dfs(int i, int j)
{
	// 如果这一行填满了
	if (j == n)
	{
		i++;
		j = 0;
	}

	// 找到一种合法的情况，就停止递归
	if (i == n) return true;
	// 如果一开始这里就有数，就看下一个格子
	if (a[i][j]) return dfs(i, j + 1);

	for (int x = 1; x <= 9; x++)
	{
		if (row[i][x] || col[j][x] || st[i / 3][j / 3][x]) continue; // 剪枝

		row[i][x] = col[j][x] = st[i / 3][j / 3][x] = true;
		a[i][j] = x;

		if (dfs(i, j + 1)) return true;

		// 恢复现场
		row[i][x] = col[j][x] = st[i / 3][j / 3][x] = false;
		a[i][j] = 0;
	}
	return false;
}

int main()
{
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			cin >> a[i][j];
			int x = a[i][j];
			if (x) row[i][x] = col[j][x] = st[i / 3][j / 3][x] = true;
		}
	}

	dfs(0, 0);

	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < n; j++)
		{
			cout << a[i][j] << ' ';
		}
		cout << endl;
	}

	return 0;
}
```
