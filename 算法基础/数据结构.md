# [【模板】单调栈](https://www.luogu.com.cn/problem/P5788)
[top](#1-单调栈)
```
#include <iostream>
#include <stack>

using namespace std;

const int N = 3e6 + 10;

int n, a[N], ans[N];

int main()
{
	cin >> n;
	for (int i = 1; i <= n; i++) cin >> a[i];

	stack<int> st;
	for (int i = n; i >= 1; i--)
	{
		while (st.size() && a[st.top()] <= a[i]) st.pop();
		if (st.size()) ans[i] = st.top();
		st.push(i);
	}

	for (int i = 1; i <= n; i++) cout << ans[i] << ' ';

	return 0;
}
```
