# [【模板】最小生成树](https://www.luogu.com.cn/problem/P3366)
[top](#1-最小生成树)  
>所有⽣成树中权值之和最⼩的树称为最⼩⽣成树
  
>Prim 算法 - 邻接矩阵
```
#include <iostream>
#include <cstring>

using namespace std;

const int N = 5010, INF = 0x3f3f3f3f;

int n, m; // n 个结点，m 条无向边
int edges[N][N]; // 邻接矩阵存储图

int dist[N]; // 某个点距离生成树的最短距离
bool st[N]; // 标记哪些点已经加入到生成树中

int prim()
{
	// 初始化
	memset(dist, 0x3f, sizeof dist);
	dist[1] = 0;

	int ret = 0;
	for (int i = 1; i <= n; i++) // 循环加入 n 个点
	{
		// 1. 找最近点
		int t = 0;
		for (int j = 1; j <= n; j++)
			if (!st[j] && dist[j] < dist[t])
				t = j;

		// 判断是否联通
		if (dist[t] == INF) return INF;
		st[t] = true;
		ret += dist[t];

		// 2. 更新距离
		for (int j = 1; j <= n; j++) // 枚举 t 能走到哪
			dist[j] = min(dist[j], edges[t][j]);
	}

	return ret;
}

int main()
{
	cin >> n >> m;

	// 初始化 邻接矩阵
	memset(edges, 0x3f, sizeof edges);

	for (int i = 1; i <= m; i++)
	{
		int x, y, z; cin >> x >> y >> z;

		// 注意重边的情况
		edges[x][y] = edges[y][x] = min(edges[x][y], z);
	}

	int ret = prim();
	if (ret == INF) cout << "orz" << endl;
	else cout << ret << endl;

	return 0;
}
```
>Prim 算法 - 邻接表 - vector 数组
```
#include <iostream>
#include <vector>
#include <cstring>

using namespace std;

typedef pair<int, int> PII;

const int N = 5010, INF = 0x3f3f3f3f;

int n, m;
vector<PII> edges[N];

int dist[N];
bool st[N];

int prim()
{
	memset(dist, 0x3f, sizeof dist);
	dist[1] = 0;
	int ret = 0;

	for (int i = 1; i <= n; i++)
	{
		// 1. 找最近点
		int t = 0;
		for (int j = 1; j <= n; j++)
			if (!st[j] && dist[j] < dist[t])
				t = j;

		// 判断是否联通
		if (dist[t] == INF) return INF;
		st[t] = true;
		ret += dist[t];

		// 2. 更新距离
		for (auto& p : edges[t])
		{
			int a = p.first, b = p.second;
			// t -> a 权值是 b
			dist[a] = min(dist[a], b);
		}
	}

	return ret;
}

int main()
{
	cin >> n >> m;
	for (int i = 1; i <= m; i++)
	{
		int x, y, z; cin >> x >> y >> z;
		edges[x].push_back({ y, z });
		edges[y].push_back({ x, z });
	}

	int ret = prim();
	if (ret == INF) cout << "orz" << endl;
	else cout << ret << endl;

	return 0;
}

>Kruskal 算法
```
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 5010, M = 2e5 + 10, INF = 0x3f3f3f3f;

int n, m;
struct node
{
	int x, y, z;
}a[M];

int fa[N]; // 并查集

bool cmp(node& a, node& b)
{
	return a.z < b.z;
}

int find(int x)
{
	return fa[x] == x ? x : fa[x] = find(fa[x]);
}

int kruskal()
{
	sort(a + 1, a + 1 + m, cmp);

	int cnt = 0;
	int ret = 0;
	for (int i = 1; i <= m; i++)
	{
		int x = a[i].x, y = a[i].y, z = a[i].z;

		int fx = find(x), fy = find(y);
		if (fx != fy)
		{
			cnt++;
			ret += z;
			fa[fx] = fy;
		}
	}

	return cnt == n - 1 ? ret : INF;
}

int main()
{
	cin >> n >> m;
	for (int i = 1; i <= m; i++) cin >> a[i].x >> a[i].y >> a[i].z;

	// 初始化并查集
	for (int i = 1; i <= n; i++) fa[i] = i;

	int ret = kruskal();
	if (ret == INF) cout << "orz" << endl;
	else cout << ret << endl;

	return 0;
}

# [买礼物](https://www.luogu.com.cn/problem/P1194)
[top](#1-最小生成树)
```
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 510;

int a, b;

int pos;
struct node
{
	int x, y, z;
}e[N * N];

int fa[N];

int find(int x)
{
	return fa[x] == x ? x : fa[x] = find(fa[x]);
}

bool cmp(node& a, node& b)
{
	return a.z < b.z;
}

int kk()
{
	int s = 0; // 统计边的权值和
	int cnt = 0; // 统计边的个数

	sort(e + 1, e + 1 + pos, cmp);

	for (int i = 1; i <= pos; i++)
	{
		int x = e[i].x, y = e[i].y, z = e[i].z;

		int fx = find(x), fy = find(y);
		if (fx == fy) continue;
		fa[fx] = fy;
		s += z;
		cnt++;
	}

	return s + (b - cnt) * a;
}

int main()
{
	cin >> a >> b;
	for (int i = 1; i <= b; i++)
	{
		for (int j = 1; j <= b; j++)
		{
			int k; cin >> k;
			if (i >= j || !k || k > a) continue;
			pos++;
			e[pos].x = i, e[pos].y = j, e[pos].z = k;
		}
	}

	for (int i = 1; i <= b; i++) fa[i] = i;

	cout << kk() << endl;

	return 0;
}

# [[SCOI2012] 滑雪](https://www.luogu.com.cn/problem/P2573)
[top](#1-最小生成树)
```
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long LL;
typedef pair<int, int> PII;

const int N = 1e5 + 10, M = 1e6 + 10;

int n, m;
int h[N];

vector<PII> edges[N]; // 存图

int cnt; // 统计最多能到达的景点
LL ans; // 最短滑行距离总和
bool st[N]; // 记录哪些景点可以到达

int pos;
struct node
{
	int x, y, z;
}e[M];

int fa[N];

int find(int x)
{
	return fa[x] == x ? x : fa[x] = find(fa[x]);
}

void dfs(int u)
{
	cnt++;
	st[u] = true;

	for (auto& p : edges[u])
	{
		int v = p.first, k = p.second;
		pos++;
		e[pos].x = u, e[pos].y = v, e[pos].z = k;
		if (!st[v]) dfs(v);
	}
}

bool cmp(node& a, node& b)
{
	int y1 = a.y, z1 = a.z, y2 = b.y, z2 = b.z;
	if (h[y1] != h[y2]) return h[y1] > h[y2]; // 因为是从高处往地处走，所以先选高的
	return z1 < z2; // 再选近的
}

void kk()
{
	for (int i = 1; i <= n; i++) fa[i] = i;

	sort(e + 1, e + 1 + pos, cmp);

	for (int i = 1; i <= pos; i++)
	{
		int x = e[i].x, y = e[i].y, z = e[i].z;
		int fx = find(x), fy = find(y);
		if (fx == fy) continue;
		fa[fx] = fy;
		ans += z;
	}
}

int main()
{
	cin >> n >> m;
	for (int i = 1; i <= n; i++) cin >> h[i];

	for (int i = 1; i <= m; i++)
	{
		int u, v, k; cin >> u >> v >> k;
		if (h[u] >= h[v]) edges[u].push_back({ v, k });
		if (h[u] <= h[v]) edges[v].push_back({ u, k });
	}

	dfs(1);
	cout << cnt << ' ';

	kk();
	cout << ans << endl;

	return 0;
}
```
# [【模板】拓扑排序 / 家谱树](https://www.luogu.com.cn/problem/B3644)
[top](#2-拓扑排序)
```
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

const int N = 110;

int n;
vector<int> edges[N];
int in[N]; // 存入度信息

int main()
{
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		int j;
		while (cin >> j, j)
		{
			edges[i].push_back(j);
			in[j]++;
		}
	}

	// 拓扑排序
	queue<int> q;
	for (int i = 1; i <= n; i++)
	{
		if (!in[i]) q.push(i);
	}

	while (q.size())
	{
		int x = q.front(); q.pop();
		cout << x << ' ';

		// 删除对应的边
		for (int y : edges[x])
		{
			in[y]--;
			if (!in[y]) q.push(y);
		}
	}

	return 0;
}

# [最大食物链计数](https://www.luogu.com.cn/problem/P4017)
[top](#2-拓扑排序)
```
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

const int N = 5010, MOD = 80112002;

int n, m;
vector<int> edges[N];
int in[N], out[N];
int f[N];

int main()
{
	cin >> n >> m;
	while (m--)
	{
		int a, b; cin >> a >> b;
		edges[a].push_back(b);
		in[b]++;
		out[a]++;
	}

	queue<int> q;
	for (int i = 1; i <= n; i++)
	{
		if (!in[i])
		{
			q.push(i);
			f[i] = 1; // 初始化生产者
		}
	}

	while (q.size())
	{
		int x = q.front(); q.pop();
		for (int y : edges[x])
		{
			in[y]--;
			f[y] = (f[y] + f[x]) % MOD;
			if (!in[y]) q.push(y);
		}
	}

	int ans = 0;
	for (int i = 1; i <= n; i++)
	{
		// 出度为零，才是最右端
		if (!out[i]) ans = (ans + f[i]) % MOD;
	}

	cout << ans << endl;

	return 0;
}

# [杂务](https://www.luogu.com.cn/problem/P1113)
[top](#2-拓扑排序)
```
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

const int N = 1e4 + 10;

int n;
int len[N];
vector<int> edges[N];
int in[N];
int f[N];

int main()
{
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		int x, y;
		cin >> y;
		cin >> len[y];
		while (cin >> x, x)
		{
			edges[x].push_back(y);
			in[y]++;
		}
	}

	queue<int> q;
	for (int i = 1; i <= n; i++)
	{
		if (!in[i]) q.push(i);
	}

	int ans = 0;

	while (q.size())
	{
		int x = q.front(); q.pop();
		f[x] += len[x];
		ans = max(ans, f[x]);

		for (int y : edges[x])
		{
			f[y] = max(f[y], f[x]);
			in[y]--;
			if (!in[y]) q.push(y);
		}
	}

	cout << ans << endl;

	return 0;
}

# [【模板】单源最短路径（弱化版）](https://www.luogu.com.cn/problem/P3371)
[top](#3-单源最短路)  
>常规版 dijkstra 算法
```
#include <iostream>
#include <climits>
#include <vector>

using namespace std;

typedef pair<int, int> PII;

const int N = 1e4 + 10, INF = INT_MAX;

int n, m, s;
vector<PII> edges[N];
int dist[N];
bool st[N];

void dijkstra()
{
	// 初始化
	for (int i = 0; i <= n; i++) dist[i] = INF;
	dist[s] = 0;

	for (int i = 1; i < n; i++) // 确定 n - 1 个点，即可遍历完所有边，得到所有点的最短路径长度
	{
		// 在没有确定的点中，找到最短的
		int u = 0;
		for (int j = 1; j <= n; j++)
			if (!st[j] && dist[j] < dist[u])
				u = j;

		st[u] = true; // 这个点已被确定了

		for (auto& t : edges[u])
		{
			int v = t.first, w = t.second;
			dist[v] = min(dist[v], dist[u] + w);
		}
	}

	for (int i = 1; i <= n; i++) cout << dist[i] << ' ';
}

int main()
{
	cin >> n >> m >> s;
	while (m--)
	{
		int u, v, w; cin >> u >> v >> w;
		edges[u].push_back({ v, w });
	}

	dijkstra();

	return 0;
}
```
>Bellman‒Ford 算法
```
#include <iostream>
#include <vector>
#include <climits>

using namespace std;

typedef pair<int, int> PII;

const int N = 1e4 + 10, INF = INT_MAX;

int n, m, s;
vector<PII> edges[N];
int dist[N];

void bf()
{
	for (int i = 1; i <= n; i++) dist[i] = INF;
	dist[s] = 0;

	bool flag = true;
	while (flag)
	{
		flag = false;
		for (int u = 1; u <= n; u++)
		{
			if (dist[u] == INF) continue;
			for (auto& t : edges[u])
			{
				int v = t.first, w = t.second;
				if (dist[u] + w < dist[v])
				{
					dist[v] = dist[u] + w;
					flag = true;
				}
			}
		}
	}

	for (int i = 1; i <= n; i++) cout << dist[i] << ' ';
}

int main()
{
	cin >> n >> m >> s;
	while (m--)
	{
		int u, v, w; cin >> u >> v >> w;
		edges[u].push_back({ v, w });
	}

	bf();

	return 0;
}

>spfa 算法
```
#include <iostream>
#include <vector>
#include <queue>
#include <climits>

using namespace std;

typedef pair<int, int> PII;

const int N = 1e4 + 10, INF = INT_MAX;

int n, m, s;
vector<PII> edges[N];
int dist[N];
bool st[N];

void spfa()
{
	for (int i = 1; i <= n; i++) dist[i] = INF;
	dist[s] = 0;

	queue<int> q;
	q.push(s);
	st[s] = true;

	while (q.size())
	{
		int u = q.front(); q.pop();
		st[u] = false;
		for (auto& t : edges[u])
		{
			int v = t.first, w = t.second;

			{

				if (!st[v])
				{
					q.push(v);
					st[v] = true;
				}
			}
		}
	}

	for (int i = 1; i <= n; i++) cout << dist[i] << ' ';
}

int main()

	cin >> n >> m >> s;
	while (m--)
	{
		int u, v, w; cin >> u >> v >> w;

# [【模板】单源最短路径（标准版）](https://www.luogu.com.cn/problem/P4779)
[top](#3-单源最短路)  
>堆优化版 dijkstra 算法
```
#include <iostream>
#include <cstring>
#include <vector>
#include <queue>

using namespace std;

typedef pair<int, int> PII;

const int N = 1e5 + 10;

int n, m, s;
vector<PII> edges[N];

bool st[N];
int dist[N];

void dijkstra()
{
	memset(dist, 0x3f, sizeof dist);
	priority_queue<PII, vector<PII>, greater<PII>> heap; // <距离, 结点>

	heap.push({ 0, s });
	dist[s] = 0;

	while (heap.size())
	{
		auto t = heap.top(); heap.pop();
		int u = t.second;

		if (st[u]) continue;
		st[u] = true;

		for (auto& t : edges[u])
		{
			int v = t.first, w = t.second;
			if (dist[u] + w < dist[v])
			{
				dist[v] = dist[u] + w;
				heap.push({ dist[v], v }); // 把新的值放进堆
			}
		}
	}

	for (int i = 1; i <= n; i++) cout << dist[i] << ' ';
}

int main()
{
	cin >> n >> m >> s;
	while (m--)
	{
		int u, v, w; cin >> u >> v >> w;
		edges[u].push_back({ v, w });
	}

	dijkstra();

	return 0;
}
```
# [【模板】负环](https://www.luogu.com.cn/problem/P3385)
[top](#3-单源最短路)  
>BF 算法判断负环：执⾏ n 轮松弛操作，如果第 n 轮还存在松弛操作，那么就有负环。 

码一：
```
#include <iostream>
#include <cstring>
#include <vector>

using namespace std;

typedef pair<int, int> PII;

const int N = 2e3 + 10;

int n, m;
vector<PII> edges[N];
int dist[N];

bool bf()
{
	memset(dist, 0x3f, sizeof dist);
	dist[1] = 0;

	for (int i = 1; i <= n; i++)
	{
		bool flag = false;
		for (int u = 1; u <= n; u++) // 如果没有负环，最多 n - 1 次结束
		{

>spfa 算法判断负环
```
#include <iostream>
#include <cstring>
#include <vector>
#include <queue>

using namespace std;

typedef pair<int, int> PII;

const int N = 2e3 + 10;

int n, m;
vector<PII> edges[N];
int dist[N];
bool st[N];
int cnt[N];

bool spfa()
{
	memset(dist, 0x3f, sizeof dist);
	memset(st, 0, sizeof st);
	memset(cnt, 0, sizeof cnt);
	dist[1] = 0;

	queue<int> q;
	q.push(1);
	st[1] = true;

	while (q.size())
	{
		int u = q.front(); q.pop();
		st[u] = false;
		for (auto& t : edges[u])
		{
			int v = t.first, w = t.second;
			if (dist[u] + w < dist[v])
			{
				dist[v] = dist[u] + w;
				cnt[v] = cnt[u] + 1;
				if (cnt[v] == n) return true;
				if (!st[v])
				{
					q.push(v);
					st[v] = true;
				}
			}
		}
		
	}
	return false;
}

int main()
{
	int T; cin >> T;
	while (T--)
	{
		for (int i = 1; i <= n; i++) edges[i].clear();
		cin >> n >> m;
		while (m--)
		{
			int u, v, w; cin >> u >> v >> w;
			edges[u].push_back({ v, w });
			if (w >= 0) edges[v].push_back({ u, w });
		}

		if (spfa()) cout << "YES" << endl;
		else cout << "NO" << endl;
	}

	return 0;
}
